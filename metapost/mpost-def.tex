\begin{mpostdef}

  color slash;

  def coordtwo(expr O, len, sx, sy) =
    drawarrow ((-0.1len,0)--(0.9len,0)) shifted O;
    drawarrow ((0,-0.1len)--(0,0.9len)) shifted O;
  enddef;

  def hatchfill expr p =
    fill p
  enddef;
